## 1. 문제

https://programmers.co.kr/learn/courses/30/lessons/12909

## 2. 풀이

~~~python
def solution(s):
    left = 0
    for i, x in enumerate(s): # 모든 글자를 확인하면서 ( 보다 ) 가 많이 나온 경우에 False return
        if x == '(': 
            left += 1
        else:
            left -= 1
            if left < 0:
                return False
                
    if left != 0: # 모든 문자열을 확인했는데 갯수가 동일하지 않으면(left != 0) False return
        return False
    return True
~~~
